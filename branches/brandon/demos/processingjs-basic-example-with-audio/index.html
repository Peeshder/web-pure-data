// Global variables
float radius = 50.0;
int X, Y;
int delay = 16;

// Setup the Processing Canvas
void setup(){
  size( 200, 200 );
  strokeWeight( 10 );
  frameRate( 15 );
  X = width / 2;
  Y = width / 2;
  pd = new Pd(44100, 4410);
  pd.load("patch.pd", pd.play);
}

// Main draw loop
void draw(){
  
  radius = radius + sin( frameCount / 4 );
  
  // Fill canvas grey
  background( 100 );
  
  // Set fill-color to blue
  fill( 0, Y, X );
  
  // Set stroke-color white
  stroke(255); 
  
  // Draw circle
  ellipse( X, Y, radius, radius );                  
  
  pd.send("x1", X);
  pd.send("y1", Y);
  pd.send("radius", radius - 50);
}


// Set circle's next destination
void mouseMoved(){
  X = mouseX;
  Y = mouseY;  
}
